## Request

Create a new accounting period for transaction posting.

### Headers

<ParamField header="Authorization" type="string" required>
  JWT authentication token. Format: `jwt YOUR_TOKEN`
</ParamField>

<ParamField header="Content-Type" type="string" required>
  Must be `application/json`
</ParamField>

### Body Parameters

<ParamField body="period_name" type="string" required>
  Display name for the period (e.g., "January 2024")
</ParamField>

<ParamField body="fiscal_year_id" type="string" required>
  Associated fiscal year identifier
</ParamField>

<ParamField body="start_date" type="string" required>
  Period start date (YYYY-MM-DD format)
</ParamField>

<ParamField body="end_date" type="string" required>
  Period end date (YYYY-MM-DD format)
</ParamField>

<ParamField body="quarter" type="number" required>
  Fiscal quarter (1-4)
</ParamField>

<ParamField body="is_adjustment_period" type="boolean" default="false">
  Whether this is an adjustment period (period 13)
</ParamField>

<ParamField body="is_closed" type="boolean" default="false">
  Whether the period should be created in closed state
</ParamField>

    "period_name": "January 2024",
