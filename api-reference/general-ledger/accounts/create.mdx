## Request

Create a new account in the chart of accounts.

### Headers

<ParamField header="Authorization" type="string" required>
  JWT authentication token. Format: `jwt YOUR_TOKEN`
</ParamField>

<ParamField header="Content-Type" type="string" required>
  Must be `application/json`
</ParamField>

### Body Parameters

<ParamField body="account_id" type="string" required>
  Unique identifier for the account
</ParamField>

<ParamField body="account_number" type="number" required>
  Numeric account number for sorting and display
</ParamField>

<ParamField body="account_name" type="string" required>
  Display name for the account
</ParamField>

<ParamField body="account_type" type="string" required>
  Account classification. One of: `Asset`, `Liability`, `Equity`, `Revenue`, `Expense`
</ParamField>

<ParamField body="normal_balance" type="string" required>
  Normal balance side. One of: `debit`, `credit`
</ParamField>

<ParamField body="currency" type="string" required>
  ISO 4217 currency code (e.g., `USD`, `EUR`)
</ParamField>

<ParamField body="balance" type="number" default="0">
  Current account balance
</ParamField>

<ParamField body="opening_balance" type="number" default="0">
  Opening balance for the account
</ParamField>

<ParamField body="is_summary" type="boolean" default="false">
  Whether this is a summary/parent account
</ParamField>

<ParamField body="is_inactive" type="boolean" default="false">
  Whether the account is inactive
</ParamField>
