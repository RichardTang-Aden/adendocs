---

## Request

Create a new approval rule for automated workflow routing.

### Body Parameters

<ParamField body="rule_id" type="string" required>
  Unique identifier for the rule
</ParamField>

<ParamField body="rule_name" type="string" required>
  Human-readable name for the rule
</ParamField>

<ParamField body="document_type" type="string" required>
  Document type: `journal_entry`, `bill`, `invoice`, `bill_payment`, `invoice_payment`, `credit_memo`, `expense`
</ParamField>

<ParamField body="rule_type" type="string" required>
  Rule type: `Amount Threshold`, `Department Routing`, `Cost Center Routing`
</ParamField>

<ParamField body="is_active" type="boolean" required>
  Whether the rule is active
</ParamField>

<ParamField body="priority" type="integer" required>
  Priority order (lower numbers execute first)
</ParamField>

<ParamField body="min_amount" type="number">
  Minimum amount threshold (for Amount Threshold rules)
</ParamField>

<ParamField body="max_amount" type="number">
  Maximum amount threshold (for Amount Threshold rules). Null for no upper limit.
</ParamField>

<ParamField body="approval_level" type="integer" required>
  Required approval level (1, 2, 3, etc.)
</ParamField>

<ParamField body="approver_type" type="string" required>
  Approver type: `role`, `user`, `manager`
</ParamField>

<ParamField body="approver_role" type="string">
  Role required for approval (when approver_type is `role`)
</ParamField>

<ParamField body="approver_user_id" type="string">
  Specific user ID (when approver_type is `user`)
</ParamField>

  }'
