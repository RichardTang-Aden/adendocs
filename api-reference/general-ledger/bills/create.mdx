## Request

Create a new vendor bill. The bill is created in Draft status.

### Headers

<ParamField header="Authorization" type="string" required>
  JWT authentication token. Format: `jwt YOUR_TOKEN`
</ParamField>

<ParamField header="Content-Type" type="string" required>
  Must be `application/json`
</ParamField>

### Body Parameters

<ParamField body="bill_id" type="string" required>
  Unique identifier for the bill
</ParamField>

<ParamField body="vendor_id" type="string" required>
  Vendor/supplier ID
</ParamField>

<ParamField body="ap_account_id" type="string" required>
  Accounts Payable GL account ID
</ParamField>

<ParamField body="tran_date" type="string" required>
  Bill date (YYYY-MM-DD)
</ParamField>

<ParamField body="due_date" type="string" required>
  Payment due date (YYYY-MM-DD)
</ParamField>

<ParamField body="posting_period_id" type="string" required>
  Accounting period ID
</ParamField>

<ParamField body="currency" type="string" default="USD">
  ISO 4217 currency code
</ParamField>

<ParamField body="exchange_rate" type="number" default="1.0">
  Exchange rate for multi-currency bills
</ParamField>

<ParamField body="memo" type="string">
  Bill description
</ParamField>

<ParamField body="status" type="string" default="Draft">
  Initial status (typically `Draft`)
</ParamField>

<ParamField body="lines" type="array" required>
  Array of line items:
  - `line_number` (number): Sequential line number
  - `account_id` (string): Expense account ID
  - `description` (string): Line description
  - `quantity` (number): Quantity
  - `rate` (number): Unit rate
  - `amount` (number): Line amount
  - `tax_rate` (number, optional): Tax rate as decimal
  - `tax_amount` (number, optional): Tax amount
</ParamField>
